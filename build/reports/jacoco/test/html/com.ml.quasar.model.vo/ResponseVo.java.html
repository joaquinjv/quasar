package com.ml.quasar.model.vo;

public class ResponseVo {
	
	private boolean ok;
	private Object json;
	private int errorCode;
	private String message;
	
	private ResponseVo() {
	};

	public static ResponseVo ok(Object data) {
		ResponseVo r = new ResponseVo();
		r.setJson(data);
		r.setErrorCode(100);
		r.setOk(true);

		return r;
	}
	public static ResponseVo ok(Object data, String message) {
		ResponseVo r = new ResponseVo();
		r.setJson(data);
		r.setMessage(message);
		r.setErrorCode(100);
		r.setOk(true);

		return r;
	}

	public static ResponseVo ok() {
		ResponseVo r = new ResponseVo();
		r.setErrorCode(100);
		r.setOk(true);

		return r;
	}
	
	public static ResponseVo error(Object data) {
		ResponseVo r = new ResponseVo();
		r.setJson(data);
		r.setErrorCode(500);
		r.setOk(false);

		return r;
	}
	
	public static ResponseVo error(Object data, String message) {
		ResponseVo r = new ResponseVo();
		r.setJson(data);
		r.setMessage(message);
		r.setErrorCode(500);
		r.setOk(false);

		return r;
	}

	public static ResponseVo unauthorized(Object data, String message) {
		ResponseVo r = new ResponseVo();
		r.setJson(data);
		r.setMessage(message);
		r.setErrorCode(401);
		r.setOk(false);

		return r;
	}

	public static ResponseVo unauthorized(Object data) {
		ResponseVo r = new ResponseVo();
		r.setJson(data);
		r.setErrorCode(401);
		r.setOk(false);

		return r;
	}
	
	public static ResponseVo unexpectedError() {
		return error(null, "unexpected error");
	}
	
	public boolean isOk() {
		return ok;
	}

	public void setOk(boolean ok) {
		this.ok = ok;
	}

	public Object getJson() {
		return json;
	}

	public void setJson(Object json) {
		this.json = json;
	}

	public int getErrorCode() {
		return errorCode;
	}

	public void setErrorCode(int errorCode) {
		this.errorCode = errorCode;
	}

	public String getMessage() {
		return message;
	}

	public void setMessage(String message) {
		this.message = message;
	}
}
